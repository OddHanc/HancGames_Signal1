<script>
  const tg = window.Telegram.WebApp;
  tg.expand();

  let start = Date.now();
  let interactions = 0;
  const maxInteractions = 7;

  const timerEl = document.getElementById("timer");
  const hanc = document.getElementById("hanc");

  // TIMER
  setInterval(() => {
    let seconds = Math.floor((Date.now() - start) / 1000);
    timerEl.innerText = seconds;
  }, 1000);

  // TAP / HOLD DETECTION
  let holdTimeout;
  let isHold = false;

  document.body.addEventListener("touchstart", () => {
    isHold = false;
    holdTimeout = setTimeout(() => {
      isHold = true;
      handleAction("HOLD");
    }, 600);
  });

  document.body.addEventListener("touchend", () => {
    clearTimeout(holdTimeout);
    if (!isHold) handleAction("TAP");
  });

  // DESKTOP SUPPORT
  document.body.addEventListener("mousedown", () => {
    isHold = false;
    holdTimeout = setTimeout(() => {
      isHold = true;
      handleAction("HOLD");
    }, 600);
  });

  document.body.addEventListener("mouseup", () => {
    clearTimeout(holdTimeout);
    if (!isHold) handleAction("TAP");
  });

  // MAIN ACTION
  function handleAction(type) {
    if (interactions >= maxInteractions) return;

    interactions++;

    fetch("/api/action", {
      method: "POST",
      headers: {
        "Content-Type": "application/json"
      },
      body: JSON.stringify({ type })
    })
    .then(res => res.json())
    .then(data => {
      if (data.correct) {
        triggerGlitch();
      } else {
        subtleFlash();
      }
    })
    .catch(() => {});
  }

  // FEEDBACK
  function triggerGlitch() {
    hanc.classList.add("glitch");

    setTimeout(() => {
      hanc.classList.remove("glitch");
    }, 200);

    if (navigator.vibrate) {
      navigator.vibrate(40);
    }
  }

  function subtleFlash() {
    hanc.classList.add("flash");

    setTimeout(() => {
      hanc.classList.remove("flash");
    }, 100);
  }
</script>
